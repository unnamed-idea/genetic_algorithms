def abline(intercept, slope):
    """Plot a line from slope and intercept"""
    
    #ref: https://stackoverflow.com/questions/7941226/...
                            #how-to-add-line-based-on-slope-and-intercept-in-matplotlib
    axes = plt.gca()
    x_vals = np.array(axes.get_xlim())
    y_vals = intercept + slope * x_vals
    plt.plot(x_vals, y_vals, '--')

def PolyCoefficients(x, coeffs):#not used in this example, can be used to make polynomial fit
    """ Returns a polynomial for ``x`` values for the ``coeffs`` provided.

    The coefficients must be in ascending order (``x**0`` to ``x**o``).
    """
    #ref: https://stackoverflow.com/questions/37352098/...
                            #plotting-a-polynomial-using-matplotlib-and-coeffiecients
    
    o = len(coeffs)
    y = 0
    for i in range(o):
        y += coeffs[i]*x**i
    return y

def fitness(chromosome):
    
    intercept = chromosome.chromosome_[0]
    slope = chromosome.chromosome_[1]
    
    residual = 0
    
    group_1 = 0
    group_2 = 0
    
    for ind,i in enumerate(x):
        line_ans = intercept + slope*i[0]
        
        if line_ans>i[1] and y[ind]>0:
            group_1 += 1
        elif line_ans<i[1] and y[ind]==0:
            group_2 += 1
    residual = group_1+group_2
    return residual
