def schwefel(x,y):
    result = abs(838 + x*sin(abs(x)**0.5)+y*sin(abs(y)**0.5))
    return result
def custom_fit_schwefel(i):
    x = i.chromosome_[0]
    y = i.chromosome_[1]  
    result = abs(838 + x*sin(abs(x)**0.5)+y*sin(abs(y)**0.5))
    return result
